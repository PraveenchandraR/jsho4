Q9. Unique Color Shirt


    A. function Unique_Shirts (arr,N) {

  let freq=[1001];
   // intilize it with zero 
   for(let i=0;i< 1001;i++) {
     freq[i]=0;
   }
   // increase the counter of color occurs
   for(let i=0;i< N;i++) {
     freq[arr[i]]++;
   }
  let ans=0;
  
  for(let i=0;i< 1001;i++) {
    if(freq[i]===1) {
      ans++;
    }
  }
  return ans;

}

    A.  function arrayMin(arr) {
